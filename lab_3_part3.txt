1.SELECT dzial.nazwa, MAX(pensja) AS maks, MIN(pensja) AS min, AVG(pensja) AS srednia FROM pracownik INNER JOIN dzial ON (pracownik.dzial=dzial.id_dzialu) GROUP BY dzial.id_dzialu;
2.SELECT klient.pelna_nazwa, SUM(p.ilosc*p.cena) AS sum FROM klient INNER JOIN zamowienie ON klient.id_klienta=zamowienie.klient INNER JOIN pozycja_zamowienia p ON p.zamowienie=zamowienie.id_zamowienia GROUP BY klient.id_klienta ORDER BY sum DESC LIMIT 10;
3.SELECT YEAR(data_zamowienia), SUM(ilosc*cena) AS sum FROM zamowienie INNER JOIN pozycja_zamowienia p ON (p.zamowienie=zamowienie.id_zamowienia) WHERE zamowienie.status_zamowienia=5 GROUP BY YEAR(data_zamowienia) ORDER BY sum DESC;
4.SELECT s.nazwa_statusu_zamowienia, SUM(p.ilosc*p.cena) AS sum FROM pozycja_zamowienia p INNER JOIN zamowienie ON p.zamowienie=zamowienie.id_zamowienia INNER JOIN status_zamowienia s ON s.id_statusu_zamowienia=zamowienie.status_zamowienia WHERE nazwa_statusu_zamowienia='anulowane';
5.SELECT miejscowosc, COUNT(id_zamowienia) AS count, SUM(ilosc*cena) FROM adres_klienta INNER JOIN klient ON adres_klienta.klient=klient.id_klienta INNER JOIN zamowienie ON klient.id_klienta=zamowienie.klient INNER JOIN pozycja_zamowienia ON pozycja_zamowienia.zamowienie=zamowienie.id_zamowienia GROUP BY miejscowosc;
6.SELECT SUM(ilosc*cena)-SUM(cena_zakupu) AS sum FROM zamowienie INNER JOIN pozycja_zamowienia p ON (p.zamowienie=zamowienie.id_zamowienia) INNER JOIN towar ON towar.id_towaru=p.towar WHERE zamowienie.status_zamowienia=5;
7.SELECT YEAR(data_zamowienia), SUM(ilosc*cena)-SUM(cena_zakupu) AS sum FROM zamowienie INNER JOIN pozycja_zamowienia p ON (p.zamowienie=zamowienie.id_zamowienia) INNER JOIN towar ON towar.id_towaru=p.towar GROUP BY YEAR(data_zamowienia);
8.SELECT nazwa_kategori, SUM(ilosc*cena_zakupu) FROM stan_magazynowy sm INNER JOIN towar ON sm.towar=towar.id_towaru INNER JOIN kategoria ON towar.kategoria=kategoria.id_kategori GROUP BY id_kategori;
9.SELECT MONTHNAME(data_urodzenia) AS miesiac, COUNT(id_pracownika) AS Liczba_pracownikow FROM pracownik GROUP BY miesiac ORDER BY MONTH(data_urodzenia) ASC;
